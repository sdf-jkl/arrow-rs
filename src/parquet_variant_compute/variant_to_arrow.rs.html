30pub(crate) fn make_variant_to_arrow_row_builder<'a>(
31    //metadata: &BinaryViewArray,
32    path: VariantPath<'a>,
33    data_type: Option<&'a datatypes::DataType>,
34    cast_options: &'a CastOptions,
35) -> Result<Box<dyn VariantToArrowRowBuilder + 'a>> {
36    use datatypes::{
37        Float16Type, Float32Type, Float64Type, Int16Type, Int32Type, Int64Type, Int8Type,
38    };
39
40    let builder = match data_type {
41        // If no data type was requested, build an unshredded VariantArray.
42        None => VariantToBinaryVariantArrowRowBuilder::new(16).with_path(path),
43        Some(datatypes::DataType::Int8) => {
44            VariantToPrimitiveArrowRowBuilder::<Int8Type>::new(cast_options).with_path(path)
45        }
46        Some(datatypes::DataType::Int16) => {
47            VariantToPrimitiveArrowRowBuilder::<Int16Type>::new(cast_options).with_path(path)
48        }
49        Some(datatypes::DataType::Int32) => {
50            VariantToPrimitiveArrowRowBuilder::<Int32Type>::new(cast_options).with_path(path)
51        }
52        Some(datatypes::DataType::Int64) => {
53            VariantToPrimitiveArrowRowBuilder::<Int64Type>::new(cast_options).with_path(path)
54        }
55        Some(datatypes::DataType::Float16) => {
56            VariantToPrimitiveArrowRowBuilder::<Float16Type>::new(cast_options).with_path(path)
57        }
58        Some(datatypes::DataType::Float32) => {
59            VariantToPrimitiveArrowRowBuilder::<Float32Type>::new(cast_options).with_path(path)
60        }
61        Some(datatypes::DataType::Float64) => {
62            VariantToPrimitiveArrowRowBuilder::<Float64Type>::new(cast_options).with_path(path)
63        }
64        _ => {
65            return Err(ArrowError::NotYetImplemented(format!(
66                "variant_get with path={:?} and data_type={:?} not yet implemented",
67                path, data_type
68            )));
69        }
70    };
71    Ok(builder)
72}

74/// Builder for converting variant values into strongly typed Arrow arrays.
75///
76/// Useful for variant_get kernels that need to extract specific paths from variant values, possibly
77/// with casting of leaf values to specific types.
78pub(crate) trait VariantToArrowRowBuilder {
79    fn append_null(&mut self) -> Result<()>;
80
81    fn append_value(&mut self, value: &Variant<'_, '_>) -> Result<bool>;
82
83    fn finish(&mut self) -> Result<ArrayRef>;
84}
85
86/// A thin wrapper whose only job is to extract a specific path from a variant value and pass the
87/// result to a nested builder.
88struct VariantPathRowBuilder<'a, T: VariantToArrowRowBuilder> {
89    builder: T,
90    path: VariantPath<'a>,
91}
92
93trait VariantToArrowRowBuilderWithPath<'a>: VariantToArrowRowBuilder {
94    fn with_path(self, path: VariantPath<'a>) -> Box<dyn VariantToArrowRowBuilder + 'a>;
95}
96
97impl<'a, T: VariantToArrowRowBuilder + 'a> VariantToArrowRowBuilderWithPath<'a> for T {
98    fn with_path(self, path: VariantPath<'a>) -> Box<dyn VariantToArrowRowBuilder + 'a> {
99        if path.is_empty() {
100            Box::new(self)
101        } else {
102            Box::new(VariantPathRowBuilder {
103                builder: self,
104                path,
105            })
106        }
107    }
108}
109
110impl<T: VariantToArrowRowBuilder> VariantToArrowRowBuilder for VariantPathRowBuilder<'_, T> {
111    fn append_null(&mut self) -> Result<()> {
112        self.builder.append_null()
113    }
114
115    fn append_value(&mut self, value: &Variant<'_, '_>) -> Result<bool> {
116        if let Some(v) = value.get_path(&self.path) {
117            self.builder.append_value(&v)
118        } else {
119            self.builder.append_null()?;
120            Ok(false)
121        }
122    }
123    fn finish(&mut self) -> Result<ArrayRef> {
124        self.builder.finish()
125    }
126}

128/// Helper function to get a user-friendly type name
129fn get_type_name<T: ArrowPrimitiveType>() -> &'static str {
130    match std::any::type_name::<T>() {
131        "arrow_array::types::Int32Type" => "Int32",
132        "arrow_array::types::Int16Type" => "Int16",
133        "arrow_array::types::Int8Type" => "Int8",
134        "arrow_array::types::Int64Type" => "Int64",
135        "arrow_array::types::UInt32Type" => "UInt32",
136        "arrow_array::types::UInt16Type" => "UInt16",
137        "arrow_array::types::UInt8Type" => "UInt8",
138        "arrow_array::types::UInt64Type" => "UInt64",
139        "arrow_array::types::Float32Type" => "Float32",
140        "arrow_array::types::Float64Type" => "Float64",
141        "arrow_array::types::Float16Type" => "Float16",
142        _ => "Unknown",
143    }
144}

146/// Builder for converting variant values to primitive values
147struct VariantToPrimitiveArrowRowBuilder<'a, T: ArrowPrimitiveType> {
148    builder: arrow::array::PrimitiveBuilder<T>,
149    cast_options: &'a CastOptions<'a>,
150}
151
152impl<'a, T: ArrowPrimitiveType> VariantToPrimitiveArrowRowBuilder<'a, T> {
153    fn new(cast_options: &'a CastOptions<'a>) -> Self {
154        Self {
155            builder: PrimitiveBuilder::<T>::new(),
156            cast_options,
157        }
158    }
159}
160
161impl<'a, T> VariantToArrowRowBuilder for VariantToPrimitiveArrowRowBuilder<'a, T>
162where
163    T: ArrowPrimitiveType,
164    for<'m, 'v> Variant<'m, 'v>: VariantAsPrimitive<T>,
165{
166    fn append_null(&mut self) -> Result<()> {
167        self.builder.append_null();
168        Ok(())
169    }
170
171    fn append_value(&mut self, value: &Variant<'_, '_>) -> Result<bool> {
172        if let Some(v) = value.as_primitive() {
173            self.builder.append_value(v);
174            Ok(true)
175        } else {
176            if !self.cast_options.safe {
177                // Unsafe casting: return error on conversion failure
178                return Err(ArrowError::CastError(format!(
179                    "Failed to extract primitive of type {} from variant {:?} at path VariantPath([])",
180                    get_type_name::<T>(),
181                    value
182                )));
183            }
184            // Safe casting: append null on conversion failure
185            self.builder.append_null();
186            Ok(false)
187        }
188    }
189
190    fn finish(&mut self) -> Result<ArrayRef> {
191        Ok(Arc::new(self.builder.finish()))
192    }
193}

195/// Builder for creating VariantArray output (for path extraction without type conversion)
196struct VariantToBinaryVariantArrowRowBuilder {
197    builder: VariantArrayBuilder,
198}
199
200impl VariantToBinaryVariantArrowRowBuilder {
201    fn new(capacity: usize) -> Self {
202        Self {
203            builder: VariantArrayBuilder::new(capacity),
204        }
205    }
206}
207
208impl VariantToArrowRowBuilder for VariantToBinaryVariantArrowRowBuilder {
209    fn append_null(&mut self) -> Result<()> {
210        self.builder.append_null();
211        Ok(())
212    }
213
214    fn append_value(&mut self, value: &Variant<'_, '_>) -> Result<bool> {
215        // TODO: We need a way to convert a Variant directly to bytes. In particular, we want to
216        // just copy across the underlying value byte slice of a `Variant::Object` or
217        // `Variant::List`, without any interaction with a `VariantMetadata` (because the shredding
218        // spec requires us to reuse the existing metadata when unshredding).
219        //
220        // One could _probably_ emulate this with parquet_variant::VariantBuilder, but it would do a
221        // lot of unnecessary work and would also create a new metadata column we don't need.
222        self.builder.append_variant(value.clone());
223        Ok(true)
224    }
225
226    fn finish(&mut self) -> Result<ArrayRef> {
227        // VariantArrayBuilder::build takes ownership, so we need to replace it
228        let builder = std::mem::replace(&mut self.builder, VariantArrayBuilder::new(0));
229        Ok(Arc::new(builder.build()))
230    }
