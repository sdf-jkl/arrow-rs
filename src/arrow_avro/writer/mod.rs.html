18//! Avro writer implementation for the `arrow-avro` crate.
19//!
20//! # Overview
21//!
22//! *   Use **`AvroWriter`** (Object Container File) when you want a
23//!     self‑contained Avro file with header, schema JSON, optional compression,
24//!     blocks, and sync markers.
25//! *   Use **`AvroStreamWriter`** (raw binary stream) when you already know the
26//!     schema out‑of‑band (i.e., via a schema registry) and need a stream
27//!     of Avro‑encoded records with minimal framing.
28//!
29
30/// Encodes `RecordBatch` into the Avro binary format.
31pub mod encoder;
32/// Logic for different Avro container file formats.
33pub mod format;

45/// Builder to configure and create a `Writer`.
46#[derive(Debug, Clone)]
47pub struct WriterBuilder {
48    schema: Schema,
49    codec: Option<CompressionCodec>,
50    capacity: usize,
51}
52
53impl WriterBuilder {
54    /// Create a new builder with default settings.
55    pub fn new(schema: Schema) -> Self {
56        Self {
57            schema,
58            codec: None,
59            capacity: 1024,
60        }
61    }
62
63    /// Change the compression codec.
64    pub fn with_compression(mut self, codec: Option<CompressionCodec>) -> Self {
65        self.codec = codec;
66        self
67    }
68
69    /// Sets the capacity for the given object and returns the modified instance.
70    pub fn with_capacity(mut self, capacity: usize) -> Self {
71        self.capacity = capacity;
72        self
73    }
74
75    /// Create a new `Writer` with specified `AvroFormat` and builder options.
76    /// Performs one‑time startup (header/stream init, encoder plan).
77    pub fn build<W, F>(self, mut writer: W) -> Result<Writer<W, F>, ArrowError>
78    where
79        W: Write,
80        F: AvroFormat,
81    {
82        let mut format = F::default();
83        let avro_schema = match self.schema.metadata.get(SCHEMA_METADATA_KEY) {
84            Some(json) => AvroSchema::new(json.clone()),
85            None => AvroSchema::try_from(&self.schema)?,
86        };
87        let mut md = self.schema.metadata().clone();
88        md.insert(
89            SCHEMA_METADATA_KEY.to_string(),
90            avro_schema.clone().json_string,
91        );
92        let schema = Arc::new(Schema::new_with_metadata(self.schema.fields().clone(), md));
93        format.start_stream(&mut writer, &schema, self.codec)?;
94        let avro_root = AvroFieldBuilder::new(&avro_schema.schema()?).build()?;
95        let encoder = RecordEncoderBuilder::new(&avro_root, schema.as_ref()).build()?;
96        Ok(Writer {
97            writer,
98            schema,
99            format,
100            compression: self.codec,
101            capacity: self.capacity,
102            encoder,
103        })
104    }
105}

107/// Generic Avro writer.
108#[derive(Debug)]
109pub struct Writer<W: Write, F: AvroFormat> {
110    writer: W,
111    schema: Arc<Schema>,
112    format: F,
113    compression: Option<CompressionCodec>,
114    capacity: usize,
115    encoder: RecordEncoder,
116}
117
118/// Alias for an Avro **Object Container File** writer.
119pub type AvroWriter<W> = Writer<W, AvroOcfFormat>;
120/// Alias for a raw Avro **binary stream** writer.
121pub type AvroStreamWriter<W> = Writer<W, AvroBinaryFormat>;
122
123impl<W: Write> Writer<W, AvroOcfFormat> {
124    /// Convenience constructor – same as [`WriterBuilder::build`] with `AvroOcfFormat`.
125    pub fn new(writer: W, schema: Schema) -> Result<Self, ArrowError> {
126        WriterBuilder::new(schema).build::<W, AvroOcfFormat>(writer)
127    }
128
129    /// Return a reference to the 16‑byte sync marker generated for this file.
130    pub fn sync_marker(&self) -> Option<&[u8; 16]> {
131        self.format.sync_marker()
132    }
133}
134
135impl<W: Write> Writer<W, AvroBinaryFormat> {
136    /// Convenience constructor to create a new [`AvroStreamWriter`].
137    pub fn new(writer: W, schema: Schema) -> Result<Self, ArrowError> {
138        WriterBuilder::new(schema).build::<W, AvroBinaryFormat>(writer)
139    }
140}
141
142impl<W: Write, F: AvroFormat> Writer<W, F> {
143    /// Serialize one [`RecordBatch`] to the output.
144    pub fn write(&mut self, batch: &RecordBatch) -> Result<(), ArrowError> {
145        if batch.schema().fields() != self.schema.fields() {
146            return Err(ArrowError::SchemaError(
147                "Schema of RecordBatch differs from Writer schema".to_string(),
148            ));
149        }
150        match self.format.sync_marker() {
151            Some(&sync) => self.write_ocf_block(batch, &sync),
152            None => self.write_stream(batch),
153        }
154    }
155
156    /// A convenience method to write a slice of [`RecordBatch`].
157    ///
158    /// This is equivalent to calling `write` for each batch in the slice.
159    pub fn write_batches(&mut self, batches: &[&RecordBatch]) -> Result<(), ArrowError> {
160        for b in batches {
161            self.write(b)?;
162        }
163        Ok(())
164    }
165
166    /// Flush remaining buffered data and (for OCF) ensure the header is present.
167    pub fn finish(&mut self) -> Result<(), ArrowError> {
168        self.writer
169            .flush()
170            .map_err(|e| ArrowError::IoError(format!("Error flushing writer: {e}"), e))
171    }
172
173    /// Consume the writer, returning the underlying output object.
174    pub fn into_inner(self) -> W {
175        self.writer
176    }
177
178    fn write_ocf_block(&mut self, batch: &RecordBatch, sync: &[u8; 16]) -> Result<(), ArrowError> {
179        let mut buf = Vec::<u8>::with_capacity(1024);
180        self.encoder.encode(&mut buf, batch)?;
181        let encoded = match self.compression {
182            Some(codec) => codec.compress(&buf)?,
183            None => buf,
184        };
185        write_long(&mut self.writer, batch.num_rows() as i64)?;
186        write_long(&mut self.writer, encoded.len() as i64)?;
187        self.writer
188            .write_all(&encoded)
189            .map_err(|e| ArrowError::IoError(format!("Error writing Avro block: {e}"), e))?;
190        self.writer
191            .write_all(sync)
192            .map_err(|e| ArrowError::IoError(format!("Error writing Avro sync: {e}"), e))?;
193        Ok(())
194    }
195
196    fn write_stream(&mut self, batch: &RecordBatch) -> Result<(), ArrowError> {
197        self.encoder.encode(&mut self.writer, batch)
198    }
199}

233    #[test]
234    fn test_ocf_writer_generates_header_and_sync() -> Result<(), ArrowError> {
235        let batch = make_batch();
236        let buffer: Vec<u8> = Vec::new();
237        let mut writer = AvroWriter::new(buffer, make_schema())?;
238        writer.write(&batch)?;
239        writer.finish()?;
240        let out = writer.into_inner();
241        assert_eq!(&out[..4], b"Obj\x01", "OCF magic bytes missing/incorrect");
242        let trailer = &out[out.len() - 16..];
243        assert_eq!(trailer.len(), 16, "expected 16‑byte sync marker");
244        Ok(())
245    }

295    #[test]
296    fn test_roundtrip_alltypes_roundtrip_writer() -> Result<(), ArrowError> {
297        let files = [
298            "avro/alltypes_plain.avro",
299            "avro/alltypes_plain.snappy.avro",
300            "avro/alltypes_plain.zstandard.avro",
301            "avro/alltypes_plain.bzip2.avro",
302            "avro/alltypes_plain.xz.avro",
303        ];
304        for rel in files {
305            let path = arrow_test_data(rel);
306            let rdr_file = File::open(&path).expect("open input avro");
307            let mut reader = ReaderBuilder::new()
308                .build(BufReader::new(rdr_file))
309                .expect("build reader");
310            let schema = reader.schema();
311            let input_batches = reader.collect::<Result<Vec<_>, _>>()?;
312            let original =
313                arrow::compute::concat_batches(&schema, &input_batches).expect("concat input");
314            let tmp = NamedTempFile::new().expect("create temp file");
315            let out_path = tmp.into_temp_path();
316            let out_file = File::create(&out_path).expect("create temp avro");
317            let codec = if rel.contains(".snappy.") {
318                Some(CompressionCodec::Snappy)
319            } else if rel.contains(".zstandard.") {
320                Some(CompressionCodec::ZStandard)
321            } else if rel.contains(".bzip2.") {
322                Some(CompressionCodec::Bzip2)
323            } else if rel.contains(".xz.") {
324                Some(CompressionCodec::Xz)
325            } else {
326                None
327            };
328            let mut writer = WriterBuilder::new(original.schema().as_ref().clone())
329                .with_compression(codec)
330                .build::<_, AvroOcfFormat>(out_file)?;
331            writer.write(&original)?;
332            writer.finish()?;
333            drop(writer);
334            let rt_file = File::open(&out_path).expect("open roundtrip avro");
335            let mut rt_reader = ReaderBuilder::new()
336                .build(BufReader::new(rt_file))
337                .expect("build roundtrip reader");
338            let rt_schema = rt_reader.schema();
339            let rt_batches = rt_reader.collect::<Result<Vec<_>, _>>()?;
340            let roundtrip =
341                arrow::compute::concat_batches(&rt_schema, &rt_batches).expect("concat roundtrip");
342            assert_eq!(
343                roundtrip, original,
344                "Round-trip batch mismatch for file: {}",
345                rel
346            );
347        }
348        Ok(())
349    }
